import { useEffect, useState } from "react";
import "./styles.css";

export default function App() {
  const dailyGoal = 30;
  const completedTime = 30; // Change this to test different values

  const [percentage, setPercentage] = useState(0);

  useEffect(() => {
    const calculatePercentage = () => {
      let num = (completedTime / dailyGoal) * 100;
      num = Math.round(num);
      setPercentage(num);
    };

    calculatePercentage();
  }, [completedTime]);

  const getEmoji = () => {
    if (percentage === 0) return "ðŸ˜ž"; // Sad emoji for 0%
    if (percentage < 50) return "ðŸ™‚"; // Slight smile for < 50%
    if (percentage < 100) return "ðŸ˜Š"; // Happy for 50% - 99%
    return "ðŸ˜Ž"; // Cool emoji for 100%
  };

  return (
    <div className="app">
      <div className="container">
        <h2 className="title">Daily Goal Tracker ðŸŽ¯</h2>
        <div className="progress-bar">
          <div className="completed" style={{ width: `${percentage}%` }}></div>
        </div>
        <p className="progress-text">
          {percentage}% of {dailyGoal}m Completed {getEmoji()}
        </p>
      </div>
    </div>
  );
}
